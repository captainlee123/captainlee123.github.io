轨迹跟踪开发文档

一、手动部署

文件导入

引入libHYTrackSDK.a 和 include头文件夹

设置后台权限

  • Target --> Capabilities --> Background Modes --> Select "Location Updates"

设置定位权限

  • Target --> Info --> NSLocationAlwaysAndWhenInUseUsageDescription
  • Target --> Info --> NSLocationWhenInUseUsageDescription

二、使用方法

引入头文件

#import "HYLocationTrack.h"

设置跟踪模式

// 设置定时打点模式 默认1分钟
[[HYLocationTrack sharedInstance] setTrackMode:HYLocationTrackModeTiming];
// 设置跟踪模式 距离每移动x米打点
[[HYLocationTrack sharedInstance] setTrackMode:HYLocationTrackModeFollow];

开启轨迹跟踪

[[HYLocationTrack sharedInstance] startLocationTracking];

停止轨迹跟踪

[[HYLocationTrack sharedInstance] stopLocationTracking];

每次打点的Block回调

[[HYLocationTrack sharedInstance] setDidUpdateLocationBlock:^(NSDictionary * _Nonnull arg) {

 }];

一些参数配置

/**
 *  定时模式的采集频率
 *
 *  @discussion
 *  默认60s 采集一次点，但会过滤掉重复点、精度太差的点（范围10-150s）
 */
@property (nonatomic, assign) NSInteger trackFrequency;

/**
 *  跟踪精度
 *  default is kCLLocationAccuracyNearestTenMeters
 */
@property(assign, nonatomic) CLLocationAccuracy desiredAccuracy;
/*
 *  距离移动通知最小值（单位：米）
 *  default is 10
 */
@property(assign, nonatomic) CLLocationDistance distanceFilter;
